---
title: "List categories"
description: "Retrieve a paginated list of all published categories, optionally sorted and populated with their icon media."
---

Returns an array of category objects. Use the `sort` parameter to control display order, and `populate=icon` to include the full icon media object.

<Note>
  Categories are used to group menu items (e.g., Espresso, Latte, Filtrado). Sort by the `order` field to retrieve them in their intended display sequence.
</Note>

## Query parameters

<ParamField query="sort[0]" type="string" default="order:asc">
  Sort field and direction. Use `order:asc` to return categories in their configured display order.
  
  Examples: `sort[0]=order:asc`, `sort[0]=title:asc`
</ParamField>

<ParamField query="populate" type="string">
  Comma-separated list of relations to populate. Pass `icon` to include the full media object for the category icon.
  
  Example: `populate=icon`
</ParamField>

<ParamField query="pagination[page]" type="number" default="1">
  Page number (1-indexed).
</ParamField>

<ParamField query="pagination[pageSize]" type="number" default="25">
  Number of results per page. Maximum is 100.
</ParamField>

<ParamField query="filters" type="object">
  Filter categories by field values using Strapi's filter syntax.
  
  Example: `filters[title][$containsi]=espresso`
</ParamField>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token obtained from the authentication endpoint.
  
  Format: `Bearer <token>`
</ParamField>

<RequestExample>

```bash cURL
curl --request GET \
  --url 'https://api.cafeacacias.com/api/categories?sort[0]=order:asc&populate=icon' \
  --header 'Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...'
```

```javascript JavaScript
const response = await fetch(
```

</RequestExample>

## Response

<ResponseField name="data" type="object[]">
  Array of category objects.

  <Expandable title="Category object">
    <ResponseField name="id" type="number">
      Unique numeric identifier for the category.
    </ResponseField>

    <ResponseField name="documentId" type="string">
      Strapi v5 document identifier.
    </ResponseField>

    <ResponseField name="title" type="string">
      Display name of the category.
    </ResponseField>

    <ResponseField name="slug" type="string">
      URL-friendly identifier auto-generated from `title`.
    </ResponseField>

    <ResponseField name="order" type="number">
      Integer that controls the display position of this category in the menu.
    </ResponseField>

    <ResponseField name="icon" type="object | null">
      Media object for the category icon. Only present when `populate=icon` is set.

      <Expandable title="Icon media object">
        <ResponseField name="id" type="number">
          Media file ID.
        </ResponseField>
        <ResponseField name="name" type="string">
          Original file name.
        </ResponseField>
        <ResponseField name="url" type="string">
          Public URL of the image file.
        </ResponseField>
        <ResponseField name="mime" type="string">
          MIME type, e.g. `image/svg+xml`.
        </ResponseField>
        <ResponseField name="width" type="number">
          Image width in pixels.
        </ResponseField>
        <ResponseField name="height" type="number">
          Image height in pixels.
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of when the category was created.
    </ResponseField>

    <ResponseField name="updatedAt" type="string">
      ISO 8601 timestamp of the last update.
    </ResponseField>

    <ResponseField name="publishedAt" type="string">
      ISO 8601 timestamp of when the category was published. Unpublished categories are excluded from results.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination metadata.

  <Expandable title="Pagination metadata">
    <ResponseField name="pagination" type="object">
      <Expandable title="properties">
        <ResponseField name="page" type="number">
          Current page number.
        </ResponseField>
        <ResponseField name="pageSize" type="number">
          Number of results per page.
        </ResponseField>
        <ResponseField name="pageCount" type="number">
          Total number of pages.
        </ResponseField>
        <ResponseField name="total" type="number">
          Total number of categories.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
