---
title: "Get a category"
description: "Retrieve a single category by its numeric ID, optionally populated with its icon media."
---

Returns a single category object. Use `populate=icon` to include the full icon media object in the response.

## Path parameters

<ParamField path="id" type="number" required>
  The numeric ID of the category to retrieve.
</ParamField>

## Query parameters

<ParamField query="populate" type="string">
  Comma-separated list of relations to populate. Pass `icon` to include the full media object for the category icon.

  Example: `populate=icon`
</ParamField>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token obtained from the authentication endpoint.

  Format: `Bearer <token>`
</ParamField>

<RequestExample>

```bash cURL
curl --request GET \
  --url 'https://api.cafeacacias.com/api/categories/3?populate=icon' \
  --header 'Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...'
```

```javascript JavaScript
const response = await fetch(
  'https://api.cafeacacias.com/api/categories/3?populate=icon',
  {
    headers: {
      Authorization: 'Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...'
    }
  }
);

const { data } = await response.json();
```

</RequestExample>

## Response

<ResponseField name="data" type="object">
  The requested category object.

  <Expandable title="Category object" defaultOpen>
    <ResponseField name="id" type="number">
      Unique numeric identifier for the category.
    </ResponseField>

    <ResponseField name="documentId" type="string">
      Strapi v5 document identifier.
    </ResponseField>

    <ResponseField name="title" type="string">
      Display name of the category.
    </ResponseField>

    <ResponseField name="slug" type="string">
      URL-friendly identifier auto-generated from `title`.
    </ResponseField>

    <ResponseField name="order" type="number">
      Integer that controls the display position of this category in the menu.
    </ResponseField>

    <ResponseField name="icon" type="object | null">
      Media object for the category icon. Only present when `populate=icon` is set.

      <Expandable title="Icon media object">
        <ResponseField name="id" type="number">
          Media file ID.
        </ResponseField>
        <ResponseField name="name" type="string">
          Original file name.
        </ResponseField>
        <ResponseField name="url" type="string">
          Public URL of the image file.
        </ResponseField>
        <ResponseField name="mime" type="string">
          MIME type, e.g. `image/svg+xml`.
        </ResponseField>
        <ResponseField name="width" type="number">
          Image width in pixels.
        </ResponseField>
        <ResponseField name="height" type="number">
          Image height in pixels.
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of when the category was created.
    </ResponseField>

    <ResponseField name="updatedAt" type="string">
      ISO 8601 timestamp of the last update.
    </ResponseField>

    <ResponseField name="publishedAt" type="string">
      ISO 8601 timestamp of when the category was published.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Empty object for single-resource responses.
</ResponseField>
