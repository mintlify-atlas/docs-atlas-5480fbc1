---
title: "List wishlist"
description: "Retrieve all wishlist entries for the current user, with optional product population."
api: "GET https://api.cafeacacias.com/api/wishlists"
---

<Note>
  Always filter by the current user's ID. Without the `filters[user][id][$eq]` parameter, Strapi returns wishlist entries for all users (subject to role permissions).
</Note>

## Authentication

<ParamField header="Authorization" type="string" required>
  Bearer token. Format: `Bearer <jwt_token>`
</ParamField>

## Query parameters

### Filtering

<ParamField query="filters[user][id][$eq]" type="number" required>
  Filter wishlist entries by user ID. Pass the authenticated user's ID to return only their wishlist.
</ParamField>

### Population

<ParamField query="populate" type="string">
  Pass `populate=product` to include the related product object in each entry.
</ParamField>

<ParamField query="populate[product][populate]" type="string">
  Nested population for the product relation. Pass `populate[product][populate]=cover` to include the product's cover image.
</ParamField>

### Pagination

<ParamField query="pagination[page]" type="number" default="1">
  Page number (1-indexed).
</ParamField>

<ParamField query="pagination[pageSize]" type="number" default="25">
  Number of results per page. Maximum: 100.
</ParamField>

## Response

<ResponseField name="data" type="object[]">
  Array of wishlist entry objects.

  <Expandable title="wishlist entry object">
    <ResponseField name="id" type="number">
      Unique numeric identifier for the wishlist entry.
    </ResponseField>
    <ResponseField name="documentId" type="string">
      Unique string identifier used in Strapi v5 document lookups.
    </ResponseField>
    <ResponseField name="product" type="object">
      Related product object. Present only when `populate=product` is requested.

      <Expandable title="product object">
        <ResponseField name="id" type="number">Product ID.</ResponseField>
        <ResponseField name="documentId" type="string">Product document ID.</ResponseField>
        <ResponseField name="title" type="string">Product display name.</ResponseField>
        <ResponseField name="price" type="number">Price in Colombian pesos (COP).</ResponseField>
        <ResponseField name="discount" type="number">Discount percentage, or `null`.</ResponseField>
        <ResponseField name="summary" type="string">Short product description.</ResponseField>
        <ResponseField name="slug" type="string">URL slug.</ResponseField>
        <ResponseField name="stock" type="number">Current inventory count.</ResponseField>
        <ResponseField name="cover" type="object">
          Cover image. Present only when `populate[product][populate]=cover` is requested.

          <Expandable title="media object">
            <ResponseField name="id" type="number">Media entry ID.</ResponseField>
            <ResponseField name="url" type="string">Relative path to the uploaded file.</ResponseField>
            <ResponseField name="name" type="string">Original file name.</ResponseField>
            <ResponseField name="width" type="number">Image width in pixels.</ResponseField>
            <ResponseField name="height" type="number">Image height in pixels.</ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="createdAt" type="string">ISO 8601 timestamp of when the entry was created.</ResponseField>
    <ResponseField name="updatedAt" type="string">ISO 8601 timestamp of the most recent update.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination metadata.

  <Expandable title="meta object" defaultOpen>
    <ResponseField name="pagination" type="object">
      <Expandable title="pagination object" defaultOpen>
        <ResponseField name="page" type="number">Current page number.</ResponseField>
        <ResponseField name="pageSize" type="number">Number of items per page.</ResponseField>
        <ResponseField name="pageCount" type="number">Total number of pages.</ResponseField>
        <ResponseField name="total" type="number">Total number of matching records.</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
