---
title: "Get an order"
description: "Retrieve a single order by its numeric ID, including its state, products, shipping address, and total payment."
---

Returns a single order object. Use `populate=user` to include the associated customer object.

<Warning>
  Ensure that your application enforces ownership checks. The API will return any order by ID to any authenticated user with read permissions — your frontend should verify the order belongs to the currently authenticated user before displaying it.
</Warning>

## Path parameters

<ParamField path="id" type="number" required>
  The numeric ID of the order to retrieve.
</ParamField>

## Query parameters

<ParamField query="populate" type="string">
  Relations to populate. Pass `user` to include the associated customer object.

  Example: `populate=user`
</ParamField>

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token obtained from the authentication endpoint.

  Format: `Bearer <token>`
</ParamField>

<RequestExample>

```bash cURL
curl --request GET \
  --url 'https://api.cafeacacias.com/api/orders/18?populate=user' \
  --header 'Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...'
```

```javascript JavaScript
const response = await fetch(
  'https://api.cafeacacias.com/api/orders/18?populate=user',
  {
    headers: {
      Authorization: 'Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...'
    }
  }
);

const { data } = await response.json();
```

</RequestExample>

## Response

<ResponseField name="data" type="object">
  The requested order object.

  <Expandable title="Order object" defaultOpen>
    <ResponseField name="id" type="number">
      Unique numeric identifier for the order.
    </ResponseField>

    <ResponseField name="documentId" type="string">
      Strapi v5 document identifier.
    </ResponseField>

    <ResponseField name="totalPayment" type="number">
      Total amount charged for the order in the payment currency, rounded to the nearest integer.
    </ResponseField>

    <ResponseField name="idPayment" type="string">
      UUID generated at order creation. Used as the order reference in the Bold payment gateway.
    </ResponseField>

    <ResponseField name="state" type="string">
      Current order state. One of: `Pedido creado`, `Pago confirmado`, `Preparando tu café`, `Enviado`, `Entregado`, `Cancelado`.
    </ResponseField>

    <ResponseField name="products" type="object[]">
      Snapshot of products included in the order at purchase time.

      <Expandable title="Product item">
        <ResponseField name="id" type="number">Product ID.</ResponseField>
        <ResponseField name="title" type="string">Product name at time of purchase.</ResponseField>
        <ResponseField name="price" type="number">Unit price at time of purchase.</ResponseField>
        <ResponseField name="discount" type="number">Discount percentage applied (0–100). `0` means no discount.</ResponseField>
        <ResponseField name="quantity" type="number">Number of units ordered.</ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="addressShiping" type="object">
      Snapshot of the shipping address captured at purchase time.

      <Expandable title="Address fields">
        <ResponseField name="title" type="string">Address label (e.g., "Casa", "Oficina").</ResponseField>
        <ResponseField name="name" type="string">Recipient full name.</ResponseField>
        <ResponseField name="address" type="string">Street address.</ResponseField>
        <ResponseField name="city" type="string">City.</ResponseField>
        <ResponseField name="state" type="string">State or department.</ResponseField>
        <ResponseField name="postalCode" type="string">Postal code.</ResponseField>
        <ResponseField name="phone" type="string">Phone number without dial code.</ResponseField>
        <ResponseField name="dialCode" type="string">Country dial code, e.g. `"+57"`.</ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="user" type="object | null">
      The associated customer. Only present when `populate=user` is passed.

      <Expandable title="User object">
        <ResponseField name="id" type="number">User ID.</ResponseField>
        <ResponseField name="username" type="string">Username.</ResponseField>
        <ResponseField name="email" type="string">Email address.</ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of when the order was created.
    </ResponseField>

    <ResponseField name="updatedAt" type="string">
      ISO 8601 timestamp of the last update.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Empty object for single-resource responses.
</ResponseField>
