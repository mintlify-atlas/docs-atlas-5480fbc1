---
title: "List addresses"
description: "Retrieve all shipping addresses belonging to the current user."
api: "GET https://api.cafeacacias.com/api/addresses"
---

<Note>
  Always filter by the current user's ID. Without the `filters[user][id][$eq]` parameter, Strapi returns addresses for all users (subject to role permissions).
</Note>

## Authentication

<ParamField header="Authorization" type="string" required>
  Bearer token. Format: `Bearer <jwt_token>`
</ParamField>

## Query parameters

### Filtering

<ParamField query="filters[user][id][$eq]" type="number" required>
  Filter addresses by user ID. Pass the authenticated user's ID to return only their addresses.
</ParamField>

### Population

<ParamField query="populate" type="string">
  Relations to include in the response. Pass `populate=user` to include the related user object.
</ParamField>

### Pagination

<ParamField query="pagination[page]" type="number" default="1">
  Page number (1-indexed).
</ParamField>

<ParamField query="pagination[pageSize]" type="number" default="25">
  Number of results per page. Maximum: 100.
</ParamField>

## Response

<ResponseField name="data" type="object[]">
  Array of address objects.

  <Expandable title="address object">
    <ResponseField name="id" type="number">
      Unique numeric identifier for the address.
    </ResponseField>
    <ResponseField name="documentId" type="string">
      Unique string identifier used in Strapi v5 document lookups.
    </ResponseField>
    <ResponseField name="title" type="string">
      Short label for the address, such as `Casa` or `Oficina`.
    </ResponseField>
    <ResponseField name="name" type="string">
      Full name of the recipient at this address.
    </ResponseField>
    <ResponseField name="address" type="string">
      Street address line.
    </ResponseField>
    <ResponseField name="city" type="string">
      City name.
    </ResponseField>
    <ResponseField name="state" type="string">
      Region or department (e.g., `Cundinamarca`, `Antioquia`).
    </ResponseField>
    <ResponseField name="postalCode" type="string">
      Postal code.
    </ResponseField>
    <ResponseField name="phone" type="string">
      Phone number, without the dial code prefix.
    </ResponseField>
    <ResponseField name="dialCode" type="string">
      International dial code prefix, e.g. `+57`. `null` if not set.
    </ResponseField>
    <ResponseField name="user" type="object">
      Related user object. Present only when `populate=user` is requested.

      <Expandable title="user object">
        <ResponseField name="id" type="number">User ID.</ResponseField>
        <ResponseField name="documentId" type="string">User document ID.</ResponseField>
        <ResponseField name="username" type="string">Username.</ResponseField>
        <ResponseField name="email" type="string">Email address.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="createdAt" type="string">ISO 8601 timestamp of when the address was created.</ResponseField>
    <ResponseField name="updatedAt" type="string">ISO 8601 timestamp of the most recent update.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination metadata.

  <Expandable title="meta object" defaultOpen>
    <ResponseField name="pagination" type="object">
      <Expandable title="pagination object" defaultOpen>
        <ResponseField name="page" type="number">Current page number.</ResponseField>
        <ResponseField name="pageSize" type="number">Number of items per page.</ResponseField>
        <ResponseField name="pageCount" type="number">Total number of pages.</ResponseField>
        <ResponseField name="total" type="number">Total number of matching records.</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
