---
title: "Get a product"
description: "Retrieve a single product by its numeric ID."
api: "GET https://api.cafeacacias.com/api/products/{id}"
---

## Authentication

<ParamField header="Authorization" type="string" required>
  Bearer token. Format: `Bearer <jwt_token>`
</ParamField>

## Path parameters

<ParamField path="id" type="number" required>
  Numeric ID of the product to retrieve.
</ParamField>

## Query parameters

<ParamField query="populate" type="string">
  Comma-separated list of relations to include in the response. Accepted values: `category`, `cover`, `wallpaper`, `gallery`.

  Can also be repeated as multiple query parameters: `populate=category&populate=gallery`.
</ParamField>

## Response

<ResponseField name="data" type="object">
  The requested product object.

  <Expandable title="product object" defaultOpen>
    <ResponseField name="id" type="number">
      Unique numeric identifier for the product.
    </ResponseField>
    <ResponseField name="documentId" type="string">
      Unique string identifier used in Strapi v5 document lookups.
    </ResponseField>
    <ResponseField name="title" type="string">
      Display name of the product.
    </ResponseField>
    <ResponseField name="price" type="number">
      Price in Colombian pesos (COP), without decimal places.
    </ResponseField>
    <ResponseField name="discount" type="number">
      Optional discount percentage (0–100). `null` if no discount is active.
    </ResponseField>
    <ResponseField name="summary" type="string">
      Short description of the product.
    </ResponseField>
    <ResponseField name="slug" type="string">
      URL-safe unique identifier derived from the title.
    </ResponseField>
    <ResponseField name="video" type="string">
      Optional URL to a product video. `null` if not set.
    </ResponseField>
    <ResponseField name="weight" type="number">
      Weight in grams.
    </ResponseField>
    <ResponseField name="stock" type="number">
      Current inventory count.
    </ResponseField>
    <ResponseField name="image" type="string">
      Optional URL to an external image. `null` if not set.
    </ResponseField>
    <ResponseField name="cover" type="object">
      Cover image media object. Present only when `populate=cover` is requested.

      <Expandable title="media object">
        <ResponseField name="id" type="number">Media entry ID.</ResponseField>
        <ResponseField name="url" type="string">Relative path to the uploaded file.</ResponseField>
        <ResponseField name="name" type="string">Original file name.</ResponseField>
        <ResponseField name="width" type="number">Image width in pixels.</ResponseField>
        <ResponseField name="height" type="number">Image height in pixels.</ResponseField>
        <ResponseField name="formats" type="object">Strapi-generated responsive format variants (thumbnail, small, medium, large).</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="wallpaper" type="object">
      Wallpaper image media object. Present only when `populate=wallpaper` is requested.
    </ResponseField>
    <ResponseField name="gallery" type="object[]">
      Array of gallery media objects. Present only when `populate=gallery` is requested.
    </ResponseField>
    <ResponseField name="category" type="object">
      Related category object. Present only when `populate=category` is requested.

      <Expandable title="category object">
        <ResponseField name="id" type="number">Category ID.</ResponseField>
        <ResponseField name="documentId" type="string">Category document ID.</ResponseField>
        <ResponseField name="name" type="string">Category display name.</ResponseField>
        <ResponseField name="slug" type="string">Category URL slug.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="createdAt" type="string">ISO 8601 timestamp of when the product was created.</ResponseField>
    <ResponseField name="updatedAt" type="string">ISO 8601 timestamp of the most recent update.</ResponseField>
    <ResponseField name="publishedAt" type="string">ISO 8601 timestamp of when the product was published.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Empty object for single-resource responses.
</ResponseField>

<RequestExample>

```bash cURL
curl --request GET \
  --url 'https://api.cafeacacias.com/api/products/1?populate=category&populate=cover&populate=gallery' \
  --header 'Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...'
```

```bash Without population
curl --request GET \
  --url 'https://api.cafeacacias.com/api/products/1' \
  --header 'Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...'
```

</RequestExample>
